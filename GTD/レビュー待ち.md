OR
(gtd="inbox" 
 AND (gtd_review_at = null OR (gtd_review_at != null AND gtd_review_at < date(today) - dur(1 day))))
OR
(gtd="someday" 
 AND (gtd_review_at = null OR (gtd_review_at != null AND gtd_review_at < date(today) - dur(14 day))))
